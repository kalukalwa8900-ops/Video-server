FROM node:20-slim

# Install FFmpeg + fonts (needed for text overlays)
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
      ffmpeg \
      fonts-dejavu-core \
      fontconfig && \
    fc-cache -f -v && \
    rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Install deps first (better layer caching)
COPY package*.json ./
RUN npm ci --omit=dev

# Copy source
COPY . .

# Create required directories
RUN mkdir -p uploads output temp

EXPOSE 3000

CMD ["node", "index.js"]
